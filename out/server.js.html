/**
 * @file server.js
 * @description Entry point of the backend server. Sets up middleware, routes, database connection,
 * and starts the Express server.
 */

const express = require("express");
const cors = require("cors");
const cookieParser = require("cookie-parser");

const connectDB = require("./config/db");
const authRoutes = require("./routes/authRoutes");
const suggestionRoutes = require("./routes/suggestionRoutes");
const { limiter } = require("./middleware/rateLimitMiddleware");

// Initialize scheduled tasks
require("./cronjobs/warnAndDeleteUser");

const PORT = process.env.PORT;
const app = express();

/**
 * CORS configuration to allow requests from the frontend.
 * @type {import("cors").CorsOptions}
 */
const corsOptions = {
  origin: process.env.FRONTEND_URL,
  credentials: true,
};

// Middleware setup
app.use(cors(corsOptions)); // Enables CORS with credentials support
app.use(cookieParser()); // Parses cookies from the request headers
app.use(express.json()); // Parses incoming JSON requests
app.use(express.urlencoded({ extended: true })); // Parses URL-encoded data

// Connect to MongoDB
connectDB();

// Routes
/**
 * @route /api/auth
 * @description Routes for authentication (login, signup, etc.)
 * @middleware rateLimiter
 */
app.use("/api/auth", limiter, authRoutes);

/**
 * @route /api/suggestions
 * @description Routes for submitting and fetching user suggestions
 * @middleware rateLimiter
 */
app.use("/api/suggestions", limiter, suggestionRoutes);

// Start the server
app.listen(PORT, () =>
  console.log(
    `\n~~~~~~~~~~~~~~~~~~~~~~~~~~~\nServer running on port ${PORT}\n~~~~~~~~~~~~~~~~~~~~~~~~~~~\n`
  )
);
